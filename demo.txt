    -- =================== 1. DANH MỤC CƠ BẢN (MASTER DATA) ===================
    CREATE TABLE VaiTro (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten_vai_tro NVARCHAR(50) NOT NULL UNIQUE
    );

    CREATE TABLE QuyenHan (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ma_quyen VARCHAR(50) NOT NULL UNIQUE,
        mo_ta NVARCHAR(200)
    );
    

    
    CREATE TABLE PhanQuyen (
        id_vai_tro INT NOT NULL FOREIGN KEY REFERENCES VaiTro(id),
        id_quyen INT NOT NULL FOREIGN KEY REFERENCES QuyenHan(id),
        PRIMARY KEY (id_vai_tro, id_quyen)
    );

    CREATE TABLE NhanVien (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_vai_tro INT NULL FOREIGN KEY REFERENCES VaiTro(id),
        ho_ten NVARCHAR(100) NOT NULL,
        chuc_vu NVARCHAR(50) NULL,
        bo_phan NVARCHAR(50) NULL,
        dien_thoai NVARCHAR(20) NULL,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        updated_at DATETIME NULL,
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE DoanKhach (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten_doan NVARCHAR(100) NOT NULL,
        ma_so_thue NVARCHAR(20) NULL,
        nguoi_dai_dien NVARCHAR(100) NULL,
        dien_thoai_lien_he NVARCHAR(20) NULL,
        chiet_khau DECIMAL(5,2) NOT NULL DEFAULT 0,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        updated_at DATETIME NULL,
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id),
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE KhachHang (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ho_ten NVARCHAR(100) NOT NULL,
        email NVARCHAR(100) NULL UNIQUE,
        dien_thoai NVARCHAR(20) NULL,
        cmnd_cccd NVARCHAR(20) NULL,
        loai_khach NVARCHAR(20) CHECK (loai_khach IN ('ca_nhan', 'doan', 'doanh_nghiep')) NOT NULL,
        diem_tich_luy INT NOT NULL DEFAULT 0,
        id_doan INT NULL FOREIGN KEY REFERENCES DoanKhach(id),
        ngay_dang_ky DATETIME NOT NULL DEFAULT GETDATE(),
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        updated_at DATETIME NULL,
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id),
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE KhuVuc (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten_khu_vuc NVARCHAR(100) NOT NULL,
        mo_ta NVARCHAR(MAX) NULL,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        updated_at DATETIME NULL,
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id),
        is_deleted BIT NOT NULL DEFAULT 0
    );

    -- =========== KIỂM SOÁT ĐƠN VỊ TÍNH (UoM) ===========
    CREATE TABLE DonViTinh (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten NVARCHAR(50) NOT NULL, -- 'Lon', 'Chai', 'Thùng', 'Kg', 'Vé'...
        ky_hieu NVARCHAR(10) NULL,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE SanPham (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten NVARCHAR(100) NOT NULL,
        loai_san_pham NVARCHAR(20) CHECK (loai_san_pham IN ('ve', 'thue', 'an_uong', 'luu_tru', 'khac')) NOT NULL,
        id_don_vi_co_ban INT NOT NULL FOREIGN KEY REFERENCES DonViTinh(id), -- Fix Logic: Biết don_gia là của Lon hay Thùng
        don_gia DECIMAL(15,0) NOT NULL,
        can_tao_token BIT NOT NULL DEFAULT 0,
        id_khu_vuc INT NULL FOREIGN KEY REFERENCES KhuVuc(id),
        mo_ta NVARCHAR(MAX) NULL,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        updated_at DATETIME NULL,
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id),
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE QuyDoiDonVi (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_san_pham INT NOT NULL FOREIGN KEY REFERENCES SanPham(id),
        id_don_vi_nho INT NOT NULL FOREIGN KEY REFERENCES DonViTinh(id),
        id_don_vi_lon INT NOT NULL FOREIGN KEY REFERENCES DonViTinh(id),
        ty_le_quy_doi DECIMAL(10,2) NOT NULL,
        gia_ban_rieng DECIMAL(15,0) NULL, -- Fix Logic: Mua sỉ (thùng) rẻ hơn mua lẻ (lon)
        la_don_vi_co_ban BIT NOT NULL DEFAULT 0,
        created_at DATETIME NOT NULL DEFAULT GETDATE()
    );

    -- =========== QUẢN LÝ COMBO & PHÂN BỔ DOANH THU ===========
    CREATE TABLE Combo (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten NVARCHAR(100) NOT NULL,
        gia DECIMAL(15,0) NOT NULL,
        mo_ta NVARCHAR(MAX) NULL,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        updated_at DATETIME NULL,
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id),
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE ComboChiTiet (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_combo INT NOT NULL FOREIGN KEY REFERENCES Combo(id),
        id_san_pham INT NOT NULL FOREIGN KEY REFERENCES SanPham(id),
        so_luong INT NOT NULL DEFAULT 1,
        ty_le_phan_bo DECIMAL(5,2) NOT NULL DEFAULT 0 -- Tổng cộng 100% để chia doanh thu
    );

    -- =================== 2. ĐƠN HÀNG & GIAO DỊCH ===================
    CREATE TABLE KhuyenMai (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ma_code VARCHAR(20) NOT NULL UNIQUE,
        phan_tram_giam DECIMAL(5,2) NULL,
        giam_toi_da DECIMAL(15,0) NULL,
        tien_giam_truc_tiep DECIMAL(15,0) NULL,
        ngay_het_han DATETIME NOT NULL,
        is_deleted BIT NOT NULL DEFAULT 0
    );

    CREATE TABLE DonHang (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_khach_hang INT NULL FOREIGN KEY REFERENCES KhachHang(id),
        id_doan INT NULL FOREIGN KEY REFERENCES DoanKhach(id),
        id_khuyen_mai INT NULL FOREIGN KEY REFERENCES KhuyenMai(id),
        tien_giam_gia DECIMAL(15,0) NOT NULL DEFAULT 0,
        thoi_gian DATETIME NOT NULL DEFAULT GETDATE(),
        tong_tien DECIMAL(15,0) NOT NULL,
        trang_thai NVARCHAR(20) CHECK (trang_thai IN ('cho_thanh_toan', 'da_thanh_toan', 'ghi_no_cong_ty', 'da_huy')) NOT NULL DEFAULT 'cho_thanh_toan',
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id)
    );

    CREATE TABLE ChiTietDonHang (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_don_hang INT NOT NULL FOREIGN KEY REFERENCES DonHang(id),
        id_san_pham INT NULL FOREIGN KEY REFERENCES SanPham(id),
        id_combo INT NULL FOREIGN KEY REFERENCES Combo(id),
        so_luong INT NOT NULL,
        don_gia_thuc_te DECIMAL(15,0) NOT NULL,
        CHECK (id_san_pham IS NOT NULL OR id_combo IS NOT NULL)
    );

    CREATE TABLE ViTriNgoi (
        id INT IDENTITY(1,1) PRIMARY KEY,
        hang NVARCHAR(10) NOT NULL,
        so_ghe INT NOT NULL,
        loai_ghe NVARCHAR(20) CHECK (loai_ghe IN ('thuong', 'vip')) NOT NULL,
        gia DECIMAL(15,0) NOT NULL
        -- id_khan_dai sẽ alter sau mục Trường đua
    );

    CREATE TABLE VeDienTu (
        id UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
        id_chi_tiet_don_hang INT NOT NULL FOREIGN KEY REFERENCES ChiTietDonHang(id),
        trang_thai NVARCHAR(20) CHECK (trang_thai IN ('chua_su_dung', 'da_su_dung', 'het_han')) NOT NULL DEFAULT 'chua_su_dung',
        thoi_gian_quet DATETIME NULL,
        id_khach_hang_su_dung INT NULL FOREIGN KEY REFERENCES KhachHang(id),
        id_vi_tri_ngoi INT NULL FOREIGN KEY REFERENCES ViTriNgoi(id)
    );

    -- =================== 3. CORE TÀI CHÍNH & VÍ (ACID) ===================
    CREATE TABLE ViDienTu (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_khach_hang INT NOT NULL FOREIGN KEY REFERENCES KhachHang(id) UNIQUE,
        so_du_kha_dung DECIMAL(15,0) NOT NULL DEFAULT 0,
        so_du_dong_bang DECIMAL(15,0) NOT NULL DEFAULT 0,
        RowVer ROWVERSION NOT NULL
    );

    CREATE TABLE TheRFID (
        ma_rfid NVARCHAR(50) PRIMARY KEY,
        id_vi INT NOT NULL FOREIGN KEY REFERENCES ViDienTu(id),
        trang_thai NVARCHAR(20) CHECK (trang_thai IN ('active', 'lost', 'revoked', 'pending')) NOT NULL DEFAULT 'active',
        ngay_kich_hoat DATETIME NOT NULL DEFAULT GETDATE(),
        ngay_huy DATETIME NULL
    );

    CREATE TABLE GiaoDichVi (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_vi INT NOT NULL FOREIGN KEY REFERENCES ViDienTu(id),
        loai_giao_dich NVARCHAR(20) CHECK (loai_giao_dich IN ('nap_tien', 'thanh_toan_dich_vu', 'thu_coc', 'hoan_coc', 'thu_tien_phat')) NOT NULL,
        so_tien DECIMAL(15,0) NOT NULL,
        id_don_hang_lien_quan INT NULL FOREIGN KEY REFERENCES DonHang(id),
        parent_transaction_id INT NULL FOREIGN KEY REFERENCES GiaoDichVi(id),
        thoi_gian DATETIME NOT NULL DEFAULT GETDATE(),
        HashSignature NVARCHAR(255) NULL,
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id)
    );

    CREATE TABLE PhieuThu (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_don_hang INT NULL FOREIGN KEY REFERENCES DonHang(id),
        id_giao_dich_vi INT NULL FOREIGN KEY REFERENCES GiaoDichVi(id),
        so_tien DECIMAL(15,0) NOT NULL,
        phuong_thuc NVARCHAR(20) CHECK (phuong_thuc IN ('tien_mat', 'chuyen_khoan', 'the', 'vi_dien_tu', 'momo', 'vnpay')) NOT NULL,
        ma_giao_dich_doi_tac VARCHAR(100) NULL, -- Lưu TransID của Webhook
        thoi_gian DATETIME NOT NULL DEFAULT GETDATE(),
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id),
        CHECK (id_don_hang IS NOT NULL OR id_giao_dich_vi IS NOT NULL)
    );

    CREATE TABLE PhieuChi (
        id INT IDENTITY(1,1) PRIMARY KEY,
        so_tien DECIMAL(15,0) NOT NULL,
        ly_do NVARCHAR(255) NULL,
        thoi_gian DATETIME NOT NULL DEFAULT GETDATE(),
        created_at DATETIME NOT NULL DEFAULT GETDATE(),
        created_by INT NULL FOREIGN KEY REFERENCES NhanVien(id)
    );

    -- =================== 4. DỊCH VỤ THUÊ ===================
    CREATE TABLE ThueDoChiTiet (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_don_hang INT NOT NULL FOREIGN KEY REFERENCES DonHang(id),
        id_san_pham INT NOT NULL FOREIGN KEY REFERENCES SanPham(id),
        so_luong INT NOT NULL,
        thoi_gian_bat_dau DATETIME NOT NULL,
        thoi_gian_ket_thuc DATETIME NULL,
        so_tien_coc DECIMAL(15,0) NOT NULL,
        trang_thai_coc NVARCHAR(20) CHECK (trang_thai_coc IN ('chua_hoan', 'da_hoan', 'da_phat')) NOT NULL DEFAULT 'chua_hoan',
        id_giao_dich_coc INT NULL FOREIGN KEY REFERENCES GiaoDichVi(id),
        id_giao_dich_hoan_coc INT NULL FOREIGN KEY REFERENCES GiaoDichVi(id),
        id_giao_dich_phat INT NULL FOREIGN KEY REFERENCES GiaoDichVi(id)
    );

    CREATE TABLE TuDo (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_khu_vuc INT NOT NULL FOREIGN KEY REFERENCES KhuVuc(id),
        ma_tu NVARCHAR(20) NOT NULL,
        kich_thuoc NVARCHAR(10) CHECK (kich_thuoc IN ('S','M','L')) NOT NULL,
        trang_thai NVARCHAR(20) CHECK (trang_thai IN ('trong', 'dang_thue', 'bao_tri')) NOT NULL DEFAULT 'trong'
    );

    CREATE TABLE ThueTu (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_chi_tiet_thue INT NOT NULL FOREIGN KEY REFERENCES ThueDoChiTiet(id),
        id_tu_do INT NOT NULL FOREIGN KEY REFERENCES TuDo(id),
        ma_pin NVARCHAR(10) NULL
    );

    -- =================== 5. KHÁCH SẠN ===================
    CREATE TABLE Phong (
        id INT IDENTITY(1,1) PRIMARY KEY,
        ten_phong NVARCHAR(50) NOT NULL,
        loai_phong NVARCHAR(50) NULL,
        suc_chua INT NULL,
        gia_ngay_thuong DECIMAL(15,0) NULL,
        gia_cuoi_tuan DECIMAL(15,0) NULL,
        trang_thai NVARCHAR(20) CHECK (trang_thai IN ('trong', 'da_dat', 'dang_su_dung', 'bao_tri')) NOT NULL DEFAULT 'trong',
        RowVer ROWVERSION NOT NULL
    );

    CREATE TABLE DatPhongChiTiet (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_don_hang INT NOT NULL FOREIGN KEY REFERENCES DonHang(id),
        ngay_nhan DATE NOT NULL,
        ngay_tra DATE NOT NULL,
        trang_thai NVARCHAR(20) CHECK (trang_thai IN ('da_dat', 'da_nhan', 'da_tra', 'huy')) NOT NULL DEFAULT 'da_dat'
    );

    CREATE TABLE ChiTietDatPhong (
        id INT IDENTITY(1,1) PRIMARY KEY,
        id_dat_phong_chi_tiet INT NOT NULL FOREIGN KEY REFERENCES DatPhongChiTiet(id),
        id_phong INT NOT NULL FOREIGN KEY REFERENCES Phong(id),
        don_gia_thuc_te DECIMAL(15,0) NOT NULL
    );

        thoi_gian_vao DATETIME NOT NULL DEFAULT GETDATE(),

    CREATE TABLE ThietBiTaoSong (

    CREATE TABLE LichTaoSong (
        id INT IDENTITY(1,1) PRIMARY KEY,

    -- =================== 16. STORED PROCEDURE TRUY VẤN TỔNG HỢP ===================
    CREATE PROCEDURE sp_GetChiTietDonHangToanPhan
        @id_don_hang INT
    AS
    BEGIN
        SET NOCOUNT ON;
        
        -- 1. Thông tin chung đơn hàng
        SELECT dh.*, kh.ho_ten, kh.dien_thoai
        FROM DonHang dh
        LEFT JOIN KhachHang kh ON dh.id_khach_hang = kh.id
        WHERE dh.id = @id_don_hang;

        -- 2. Sản phẩm/Combo mua đứt
        SELECT ctdh.*, ISNULL(sp.ten, cb.ten) AS ten_item
        FROM ChiTietDonHang ctdh
        LEFT JOIN SanPham sp ON ctdh.id_san_pham = sp.id
        LEFT JOIN Combo cb ON ctdh.id_combo = cb.id
        WHERE ctdh.id_don_hang = @id_don_hang;

        -- 3. Đặt phòng khách sạn
        SELECT dp.*, p.ten_phong
        FROM DatPhongChiTiet dp
        JOIN ChiTietDatPhong ctdp ON dp.id = ctdp.id_dat_phong_chi_tiet
        JOIN Phong p ON ctdp.id_phong = p.id
        WHERE dp.id_don_hang = @id_don_hang;

        -- 4. Thuê đồ (Phao, Tủ, Chòi)
        SELECT td.*, sp.ten
        FROM ThueDoChiTiet td
        JOIN SanPham sp ON td.id_san_pham = sp.id
        WHERE td.id_don_hang = @id_don_hang;

        -- 5. Bãi đỗ xe
        SELECT vdx.*, lvr.bien_so, lvr.thoi_gian_vao
        FROM VeDoXeChiTiet vdx
        JOIN LuotVaoRaBaiXe lvr ON vdx.id_luot_vao_ra = lvr.id
        WHERE vdx.id_don_hang = @id_don_hang;

        -- 6. Đặt bàn nhà hàng
        SELECT db.*, nh.ten_nha_hang
        FROM DatBan db
        JOIN NhaHang nh ON db.id_nha_hang = nh.id
        WHERE db.id_don_hang = @id_don_hang;

        -- 7. Vườn thú (Cho ăn)
        SELECT dcta.*, dv.ten AS ten_con_vat
        FROM DatChoThuAn dcta
        JOIN DongVat dv ON dcta.id_dong_vat = dv.id
        WHERE dcta.id_don_hang = @id_don_hang;
    END;
    GO

    -- =================== 17. Đảm bảo tỷ lệ phân bổ luôn 100% (oke) ===================
    CREATE TRIGGER tr_CheckComboRate
    ON ComboChiTiet
    AFTER INSERT, UPDATE
    AS
    BEGIN
        IF EXISTS (
            SELECT id_combo     
            FROM ComboChiTiet 
            GROUP BY id_combo 
            HAVING SUM(ty_le_phan_bo) <> 100
        )
        BEGIN
            RAISERROR ('Tổng tỷ lệ phân bổ của các món trong Combo phải bằng 100%', 16, 1);
            ROLLBACK TRANSACTION;
        END
    END;
    GO
